3 · Rough.js Config per Mark

import rough from 'roughjs';
const gen = rough.generator();

export const seedFor = (r: number, c: number, salt = 0) =>
  ((r*73856093) ^ (c*19349663) ^ (salt*83492791)) >>> 0;

const markOpts = (color: string, r01: number, seed: number) => ({
  stroke: color, strokeWidth: 2.1,
  roughness: 0.6 + r01 * 2.4,   // 0.6..3.0
  bowing:    1.0 + r01 * 2.0,   // 1.0..3.0
  seed,
});

// X: two diagonal strokes, 22% inner pad
export function xPaths(cx:number, cy:number, size:number, color:string, r01:number, seed:number) {
  const pad = size * 0.22;
  return [
    ...extract(gen.line(cx-size/2+pad, cy-size/2+pad, cx+size/2-pad, cy+size/2-pad, markOpts(color, r01, seed))),
    ...extract(gen.line(cx+size/2-pad, cy-size/2+pad, cx-size/2+pad, cy+size/2-pad, markOpts(color, r01, seed+1))),
  ];
}

// O: arc -0.2π → 2π-0.1 "almost closed" loop
export function oPaths(cx:number, cy:number, size:number, color:string, r01:number, seed:number) {
  const r = size * 0.32;
  return extract(gen.arc(cx, cy, r*2, r*2, -0.2*Math.PI, 2*Math.PI - 0.1, false, markOpts(color, r01, seed)));
}

// Win loop: rounded tube around line endpoints
export function winLoopPaths(points: {x:number;y:number}[], color: string, seed = 29) {
  const a = points[0], b = points[points.length-1];
  const len = Math.hypot(b.x-a.x, b.y-a.y) || 1;
  const ux = (b.x-a.x)/len, uy = (b.y-a.y)/len, nx = -uy, ny = ux;
  const e = 22;
  const verts = [
    [a.x+nx*e-ux*e, a.y+ny*e-uy*e],
    [b.x+nx*e+ux*e, b.y+ny*e+uy*e],
    [b.x-nx*e+ux*e, b.y-ny*e+uy*e],
    [a.x-nx*e-ux*e, a.y-ny*e-uy*e],
    [a.x+nx*e-ux*e, a.y+ny*e-uy*e],
  ];
  return extract(gen.curve(verts, { stroke: color, strokeWidth: 2.6, roughness: 2.0, bowing: 2.5, seed }));
}

// Win strike: line extended 10px past each endpoint
export function winStrikePaths(points: {x:number;y:number}[], color: string, seed = 31) {
  const a = points[0], b = points[points.length-1];
  const len = Math.hypot(b.x-a.x, b.y-a.y) || 1;
  const ux = (b.x-a.x)/len, uy = (b.y-a.y)/len;
  return extract(gen.line(a.x-ux*10, a.y-uy*10, b.x+ux*10, b.y+uy*10,
    { stroke: color, strokeWidth: 2.2, roughness: 1.6, bowing: 1.5, seed }));
}

// Grid line: dashed, barely wobbly
export const gridLineOpts = (color: string) => ({
  stroke: color, strokeWidth: 0.7, roughness: 0.6, bowing: 0.8,
  strokeLineDash: [2, 3] as [number, number],
});

// Extract path d from Rough drawable so we can animate stroke-dashoffset
function extract(drawable: ReturnType<typeof gen.line>): { d: string }[] {
  const out: { d: string }[] = [];
  for (const set of drawable.sets) {
    if (set.type !== 'path' && set.type !== 'fillPath') continue;
    let d = '';
    for (const op of set.ops) {
      if (op.op === 'move')     d += `M${op.data[0]} ${op.data[1]} `;
      if (op.op === 'lineTo')   d += `L${op.data[0]} ${op.data[1]} `;
      if (op.op === 'bcurveTo') d += `C${op.data.slice(0,2).join(' ')},${op.data.slice(2,4).join(' ')},${op.data.slice(4,6).join(' ')} `;
    }
    out.push({ d: d.trim() });
  }
  return out;
}

6 · Heuristic Opponent

For each empty cell within a 5×5 window of any existing mark, compute score for both "offence" (own chains, 1.0×) and "defence" (opponent chains, 0.9×), pick max.

// Per direction from candidate cell:
// count = run length if placed there (counting own chain both sides)
// opens = 0..2 (how many endpoints are still empty)
5 in a row           → 100000
4, both open         → 10000
4, one open          → 1000
3, both open         → 500
3, one open          → 80
2, both open         → 40
2, one open          → 8
else                 → 1

Sum across 4 directions, add small random tiebreaker, pick best. Difficulty scales delay only. For "easy", consider weakening defence multiplier to ~0.4.

7 · Accessibility

  • Keyboard nav. Arrow keys move focus ring; Enter / Space places.
  • Screen reader. aria-live="polite": "X placed at D7, your turn".
  • Color. X / O differ by shape, not colour — supports colourblind users in (c).
  • Reduced motion. Honour prefers-reduced-motion.
  • Contrast. In (b), 1e4a8f on fbfcfb is 7.1:1 — AAA.
